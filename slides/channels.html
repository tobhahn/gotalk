Channels package main; import ("fmt"; "time") func main() { result := make(chan string) go func() { time.Sleep(20*time.Millisecond) result <- "Done." }() select { case r := <-result: fmt.Println(r) case <-time.After(time.Second): fmt.Println("Timed out waiting.") } }
  • sending and receiving blocks by default
  • select uses first available value
  • select does not fall-through
